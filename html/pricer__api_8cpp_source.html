53// LookbackCountPoints
54// Compute how many points are needed for a grid Smin..Smax with step dS.
55
56// Return codes:
57// n>=0 = number of points
58// -1 = invalid step (dS <= 0)
59// -2 = invalid range (Smax < Smin)
60int DLL_CALL LookbackCountPoints(double Smin, double Smax, double dS) {
61 if (dS <= 0.0) return -1;
62 if (Smax < Smin) return -2;
63
64 // includiamo entrambi gli estremi come fai nel for (S0 <= Smax)
65 double span = Smax - Smin;
66 int n = (int)std::floor(span / dS + 1.0 + 1e-12);
67 return std::max(n, 0);
68}
