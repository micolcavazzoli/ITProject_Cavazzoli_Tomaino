5//Function to simulate a single path of the option lookback
6double simulate_lookback(double S0, double r, double sigma, double T, int steps, Option type, std::mt19937& gen, std::normal_distribution<>& Z) {
7
8 double dt = T / steps;
9 double S = S0;
10 double minS = S0;
11 double maxS = S0;
12
13 for (int i = 0; i < steps; i++) {
14 double z = Z(gen);
15 S *= std::exp((r - 0.5 * sigma * sigma) * dt + sigma * sqrt(dt) * z);
16 minS = std::min(minS, S);
17 maxS = std::max(maxS, S);
18 }
19 if (type == Option::Call)
20 return S - minS; //payoff call lookback
21 else
22 return maxS - S; //payoff put lookback
23}
