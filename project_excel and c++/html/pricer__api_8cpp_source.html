100// LookbackCountPoints
101// Compute how many points are needed for a grid Smin..Smax with step dS.
102
103// Return codes:
104// n>=0 = number of points
105// -1 = invalid step (dS <= 0)
106// -2 = invalid range (Smax < Smin)
107int DLL_CALL LookbackCountPoints(double Smin, double Smax, double dS) {
108 if (dS <= 0.0) return -1;
109 if (Smax < Smin) return -2;
110
111 // includiamo entrambi gli estremi come fai nel for (S0 <= Smax)
112 const double span = Smax - Smin;
113 const int n = (int)std::floor(span / dS + 1.0 + 1e-12);
114 return std::max(n, 0);
115}
