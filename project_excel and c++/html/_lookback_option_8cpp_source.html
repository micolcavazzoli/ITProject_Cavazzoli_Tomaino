5double LookbackOption::simulate_lookback(Option type, double S0, double r, double sigma, double T, int steps, std::mt19937& gen, std::normal_distribution<>& Z) const{
6
7 double dt = T / steps;
8 double S = S0;
9 double minS = S0;
10 double maxS = S0;
11
12 for (int i = 0; i < steps; i++) {
13 double z = Z(gen);
14 S *= std::exp((r - 0.5 * sigma * sigma) * dt + sigma * std::sqrt(dt) * z);
15 minS = std::min(minS, S);
16 maxS = std::max(maxS, S);
17 }
18 if (type == Option::Call)
19 return S - minS; //payoff call lookback
20 else
21 return maxS - S; //payoff put lookback
22}
